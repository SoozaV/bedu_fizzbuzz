FizzBuzz!

Introduce un valor entero entre 10 y 100 para conocer los múltiplos de 3 y 5, si el número es múltiplo de 3 aparecerá la leyenda "Fizz", si es múltiplo de 5 aparecerá la leyenda "Buzz", si el número es múltiplo de ambos aparecerá "FizzBuzz".

let valorRango = 100

for(let i = 1; i <= valorRango; i++){
    switch(true){
        case ((i % 3 == 0) && (i % 5 == 0)):
            console.log(i, "FizzBuzz")
            break
        case (i % 3 == 0):
            console.log(i, 'Fizz')
            break
        case (i % 5 == 0):
            console.log(i, "Buzz")
            break
        default:
            console.log(i)
    }
}
